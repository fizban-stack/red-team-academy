Bug Bounty Tools Reference

A curated tool reference organized by attack category — covering recon, exploitation, and miscellaneous utilities commonly used in bug bounty and web application testing. For methodology and testing approach see Bug Bounty Methodology.

Recon — Subdomain Enumeration

# Core subdomain tools:

# subfinder — passive subdomain discovery from many APIs:
go install -v github.com/projectdiscovery/subfinder/v2/cmd/subfinder@latest
subfinder -d target.com -o subs.txt
subfinder -d target.com -all -recursive -o subs.txt

# amass — in-depth attack surface mapping (passive + active):
go install -v github.com/owasp-amass/amass/v4/...@master
amass enum -passive -d target.com
amass enum -active -brute -d target.com -o subs.txt

# puredns — fast resolver with wildcard filtering:
go install github.com/d3mondev/puredns/v2@latest
puredns bruteforce /usr/share/wordlists/dns.txt target.com

# Combined pipeline:
subfinder -d target.com | httpx -silent | nuclei -t technologies/

# Permutation generation with alterx:
echo "target.com" | alterx | puredns resolve

# GitHub/GitLab subdomain discovery:
github-subdomains -d target.com -t GITHUB_TOKEN
gitlab-subdomains -d target.com -t GITLAB_TOKEN

Recon — HTTP Probing & Tech Fingerprinting

# httpx — fast HTTP multi-prober:
go install -v github.com/projectdiscovery/httpx/cmd/httpx@latest
cat subs.txt | httpx -silent -status-code -title -tech-detect
cat subs.txt | httpx -silent -mc 200,301,302 -o live.txt

# Probe for specific paths:
cat live.txt | httpx -path /admin/ -mc 200

# wafw00f — WAF detection:
pip3 install wafw00f
wafw00f https://target.com

# whatweb — technology fingerprinting:
whatweb -a 3 https://target.com

# gowitness — web screenshots at scale:
go install github.com/sensepost/gowitness@latest
gowitness scan --file live.txt --screenshot-path ./screenshots/
gowitness file -f live.txt   # screenshot all from file

# graphw00f — GraphQL engine fingerprinting:
pip3 install graphw00f
graphw00f -d -t https://target.com/graphql

Recon — Content Discovery

# ffuf — fast web fuzzer (directory, file, virtual host):
go install github.com/ffuf/ffuf/v2@latest
ffuf -w /usr/share/seclists/Discovery/Web-Content/raft-large-words.txt \
  -u https://target.com/FUZZ -mc 200,301,302,403

# Fuzz extensions:
ffuf -w words.txt -u https://target.com/FUZZ.php -mc 200

# Vhost fuzzing:
ffuf -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-20000.txt \
  -u https://target.com/ -H "Host: FUZZ.target.com" -mc 200

# feroxbuster — recursive content discovery:
cargo install feroxbuster
feroxbuster -u https://target.com -w /usr/share/seclists/Discovery/Web-Content/raft-medium-words.txt \
  -x php,html,js --depth 3

# katana — next-gen web crawler:
go install github.com/projectdiscovery/katana/cmd/katana@latest
katana -u https://target.com -d 5 -jc -o endpoints.txt

# kiterunner — API endpoint bruteforcer:
kr scan https://target.com -w routes-large.kite

# Extract JS endpoints with jsluice:
jsluice urls < app.js

Recon — Parameter Discovery

# Arjun — HTTP parameter discovery:
pip3 install arjun
arjun -u https://target.com/search
arjun -u https://target.com/api/users --post --json

# ParamSpider — mine parameters from Wayback Machine archives:
pip3 install paramspider
paramspider -d target.com

# x8 — hidden parameter discovery (Rust, fast):
cargo install x8
x8 -u https://target.com/page?FUZZ=test -w params.txt

# Combined pipeline: find params, test for reflection:
paramspider -d target.com | qsreplace "XSS_TEST" | httpx -mc 200 -mr "XSS_TEST"

# gf patterns — grep for interesting parameters from URLs:
go install github.com/tomnomnom/gf@latest
cat urls.txt | gf xss
cat urls.txt | gf sqli
cat urls.txt | gf ssrf
cat urls.txt | gf redirect

Exploitation — Command Injection

# commix — automated command injection exploitation
# github.com/commixproject/commix

pip3 install commix

# Test a URL parameter:
commix --url="http://target.com/ping?host=127.0.0.1"

# Test POST parameter:
commix --url="http://target.com/ping" --data="host=127.0.0.1"

# With cookie:
commix --url="http://target.com/ping?host=127.0.0.1" \
  --cookie="session=abc123"

# Techniques tested:
# - Classic: ; id, && id, | id
# - Blind time-based: sleep 5
# - File-based: write to /tmp, verify with read
# - Reverse shell: --reverse-shell

# Use results from Burp: save request to file
commix -r request.txt --data="host=127.0.0.1"

Exploitation — CORS Misconfiguration

# Corsy — CORS misconfiguration scanner
# github.com/s0md3v/Corsy

pip3 install corsy
python3 corsy.py -u https://target.com -h
python3 corsy.py -i urls.txt -t 10    # bulk scan with 10 threads

# What Corsy tests:
# - Reflected Origin: Origin: attacker.com → Access-Control-Allow-Origin: attacker.com
# - Null origin: Origin: null → Access-Control-Allow-Origin: null
# - Prefix/suffix bypass: Origin: noteviltarget.com
# - Subdomain bypass: Origin: evil.target.com

# CORStest — focused CORS testing:
git clone https://github.com/RUB-NDS/CORStest
python3 corstest.py https://target.com

# Manual test with curl:
curl -H "Origin: https://attacker.com" -I https://target.com/api/user
# Look for: Access-Control-Allow-Origin: https://attacker.com
#           Access-Control-Allow-Credentials: true

Exploitation — CRLF Injection

# CRLFsuite — CRLF injection scanner
# github.com/Nefcore/CRLFsuite

pip3 install crlfuzz
crlfuzz -u "https://target.com/?param=value"
crlfuzz -l urls.txt -c 50   # concurrent

# Payloads tested:
# %0d%0a (URL encoded CRLF)
# %0a%0d
# \r\n (raw)
# Various encoding bypasses

# crlfuzz (Go alternative):
go install github.com/dwisiswant0/crlfuzz@latest
crlfuzz -u "https://target.com/FUZZ" -w /path/to/crlf-payloads.txt

# What to look for in response headers:
# Injected-Header: appears in response → confirmed CRLF
# Can be used to inject: Set-Cookie, Location (redirect), custom headers

Exploitation — Web Cache Poisoning

# param-miner — find hidden, unlinked parameters (Burp extension)
# github.com/PortSwigger/param-miner

# Install via Burp BApp Store → "Param Miner"
# Right-click any request → Extensions → Param Miner → Guess params

# Burp Scanner will flag web cache poisoning if:
# - Unkeyed input affects response (cache key doesn't include param)
# - Response is cached (X-Cache: hit, Age header present)

# toxicache — Go scanner for cache poisoning:
go install github.com/xhzeem/toxicache@latest
toxicache -l urls.txt

# Manual indicators of cacheable responses:
# X-Cache: hit / miss
# CF-Cache-Status: HIT
# Age: [number]
# Cache-Control: public, max-age=...

# Common unkeyed inputs to test:
# X-Forwarded-Host, X-Original-URL, X-Rewrite-URL
# X-Forwarded-For (affects redirect URL sometimes)
# Custom headers if reflected in response

Miscellaneous — JWT Testing

# jwt_tool — JWT testing, tweaking, and cracking
# github.com/ticarpi/jwt_tool

pip3 install jwt_tool

# Decode and inspect a token:
python3 jwt_tool.py eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...

# Scan for known vulnerabilities:
python3 jwt_tool.py TOKEN -t https://target.com/api/user -rh "Authorization: Bearer TOKEN"

# Algorithm confusion (RS256 → HS256 with public key):
python3 jwt_tool.py TOKEN -X k -pk public_key.pem

# None algorithm:
python3 jwt_tool.py TOKEN -X a

# Brute-force secret (HS256):
python3 jwt_tool.py TOKEN -C -d /usr/share/wordlists/rockyou.txt

# Tamper payload (modify claims):
python3 jwt_tool.py TOKEN -T     # interactive tamper mode

# Hashcat JWT cracking:
hashcat -a 0 -m 16500 TOKEN.txt /usr/share/wordlists/rockyou.txt

Miscellaneous — 403 Bypass

# nomore403 — bypass 403/40X restrictions
# github.com/devploit/nomore403

go install github.com/devploit/nomore403@latest
nomore403 -u https://target.com/admin

# Techniques tested:
# - HTTP method override (PUT, PATCH, etc.)
# - Header injection (X-Original-URL, X-Rewrite-URL)
# - Path manipulation (//admin, /./admin, /admin/)
# - Case variation (/ADMIN, /Admin)
# - URL encoding bypasses

# Forbidden Buster — 401/403 bypass automation:
pip3 install forbidden-buster
python3 forbidden_buster.py -u https://target.com/admin

# Manual techniques:
# Add: X-Forwarded-For: 127.0.0.1
# Add: X-Original-URL: /admin
# Try: GET /admin HTTP/1.0 (HTTP/1.0 may bypass auth)
# Try: TRACE /admin (reveals middleware headers)

Useful Utilities

# tomnomnom toolkit — essential CLI utilities:
go install github.com/tomnomnom/anew@latest         # append new lines only
go install github.com/tomnomnom/gf@latest           # grep patterns
go install github.com/tomnomnom/unfurl@latest        # parse URL parts
go install github.com/tomnomnom/qsreplace@latest    # replace query string values
go install github.com/tomnomnom/waybackurls@latest  # Wayback Machine URLs

# Usage examples:
subfinder -d target.com | anew subs.txt             # deduplicate adds
cat urls.txt | gf xss | dalfox pipe                 # XSS pipeline
cat urls.txt | unfurl --unique paths                 # extract unique paths
cat urls.txt | qsreplace "' OR '1'='1" | httpx -mr "SQL error" # SQLi probe

# gau — fetch known URLs from AlienVault + Wayback + CommonCrawl:
go install github.com/lc/gau/v2/cmd/gau@latest
gau target.com | grep "\.php\|\.asp\|\.jsp" | anew endpoints.txt

# interactsh — out-of-band interaction server:
go install -v github.com/projectdiscovery/interactsh/cmd/interactsh-client@latest
interactsh-client   # generates unique URL, monitors for DNS/HTTP/SMTP callbacks
# Use URL in payloads for blind SSRF, blind XSS, XXE, etc.

# uro — deduplicate and clean URL lists:
pip3 install uro
cat urls.txt | uro   # removes duplicate paths, keeps unique param combos
